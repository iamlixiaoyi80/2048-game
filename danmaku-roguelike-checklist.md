  private config: BulletConfig;
  private flightTime: number = 0;

  constructor(
    scene: Phaser.Scene,
    x: number,
    y: number,

  // ç»˜åˆ¶å½¢çŠ¶
  private drawShape(): void {

      case 'triangle':

                  t * t * this.config.x;

        break;

        this.x += Math.cos(angle) * 2;
        this.y += Math.sin(angle) * 2;

        this.y += this.config.velocity.y * delta;
